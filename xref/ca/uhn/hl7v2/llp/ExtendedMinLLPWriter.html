27  package ca.uhn.hl7v2.llp;
28  
29  import java.io.IOException;
30  import java.io.OutputStream;
31  import java.nio.charset.Charset;
32  
33  /**
34   * ExtendedMinLLPWriter uses a message-dependent encoder to obey a
35   * potentially populated field MSH-18.
36   *
37   * @author Christian Ohr
38   */
39  public class ExtendedMinLLPWriter extends HL7EncoderWriter<ExtendedMllpEncoder> {
40  
41      public ExtendedMinLLPWriter() {
42      }
43  
44      public ExtendedMinLLPWriter(OutputStream out) throws IOException  {
45          super(out);
46      }
47  
48      public ExtendedMinLLPWriter(OutputStream out, Charset charset) throws IOException {
49          super(out, charset, false);
50      }
51  
52      public ExtendedMinLLPWriter(OutputStream out, Charset charset, boolean omitBOM) throws IOException {
53          super(out, charset, omitBOM);
54      }
55  
56      @Override
57      protected ExtendedMllpEncoder initEncoder() {
58          return new ExtendedMllpEncoder(getCharset(), omitBOM);
59      }
60  
61  }
