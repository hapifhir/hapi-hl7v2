26  package ca.uhn.hl7v2.testpanel.util;
27  
28  import java.awt.Insets;
29  import java.awt.Rectangle;
30  import java.awt.Window;
31  
32  /**
33   * Based on example given at
34   * http://stackoverflow.com/questions/1936566/how-do-you-get-the-screen-width-in-java
35   */
36  public class ScreenBoundsUtil {
37  
38  	static public Rectangle getScreenBounds(Window wnd) {
39  		Rectangle sb;
40  		Insets si = getScreenInsets(wnd);
41  
42  		sb = wnd.getGraphicsConfiguration().getBounds();
43  
44  		sb.x += si.left;
45  		sb.width -= (si.left + si.right);
46  		sb.y += si.top;
47  		sb.height -= (si.top + si.bottom);
48  		return sb;
49  	}
50  
51  	static public Insets getScreenInsets(Window wnd) {
52  		Insets si;
53  		si = wnd.getToolkit().getScreenInsets(wnd.getGraphicsConfiguration());
54  		return si;
55  	}
56  
57  }
