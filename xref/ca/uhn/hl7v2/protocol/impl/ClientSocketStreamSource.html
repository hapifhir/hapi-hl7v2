18  public class ClientSocketStreamSource extends SocketStreamSource {
19  
20      private SocketAddress myAddress;
21      private Socket mySocket;
22  
23      /**
24       * @param theAddress the address at which to connect sockets
25       * @throws TransportException
26       */
27      public ClientSocketStreamSource(SocketAddress theAddress) throws TransportException {
28          myAddress = theAddress;
29      }
30  
31      /** 
32       * @see ca.uhn.hl7v2.protocol.impl.SocketStreamSource#getSocket()
33       */
34      public Socket getSocket() {
35          return mySocket;
36      }
37  
38      /** 
39       * Gets fresh instances of sockets.  
40       */
41      public void connect() throws TransportException {
42          mySocket = getSocket(myAddress);
43      }
44  
45      private Socket getSocket(SocketAddress theAddress) throws TransportException {
46          Socket s = new Socket();        
47          try {
48              s.connect(theAddress);
49          } catch (IOException e) {
50              throw new TransportException(e);
51          }
52          return s;        
53      }
