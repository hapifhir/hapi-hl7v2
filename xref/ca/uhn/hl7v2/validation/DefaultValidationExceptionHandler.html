26  package ca.uhn.hl7v2.validation;
27  
28  import ca.uhn.hl7v2.DefaultHapiContext;
29  import ca.uhn.hl7v2.HapiContext;
30  
31  /**
32   * Simple implementation of a handler that just tracks if
33   * {@link #error(ValidationException)} has been called at least once.
34   * 
35   * @author Christian Ohr
36   */
37  public class DefaultValidationExceptionHandler extends AbstractValidationExceptionHandler<Boolean> 
38                       implements ValidationExceptionHandlerFactory<Boolean> {
39  
40  	private boolean result = true;
41  
42  	public DefaultValidationExceptionHandler() {
43          super(new DefaultHapiContext());
44      }
45  	
46      public DefaultValidationExceptionHandler(HapiContext context) {
47          super(context);
48      }
49  
50      /**
51  	 * If the validation process encounters a violation, this method is called.
52  	 * 
53  	 * @param exception a {@link ValidationException} of error severity.
54  	 */
55  	public void error(final ValidationException exception) {
56  		result = false;
57  	}
58  
59  	/**
60  	 * Called after the validation process. Should return an overall boolean validation result.
61  	 * 
62  	 * @return the overall assessment of the validation process. This method should usually return
63  	 *         <code>false</code> if {@link #onExceptions(ValidationException[])} has been
64  	 *         called at least once.
65  	 * @see {@link DefaultValidationExceptionHandler}
66  	 */
67  	public Boolean result() {
68  		return result;
69  	}
70  
71      public boolean hasFailed() {
72          return !result;
73      }
74  
75  	public ValidationExceptionHandler<Boolean> getNewInstance(HapiContext theContext) {
76  		return new DefaultValidationExceptionHandler(theContext);
77  	}
78  	
79  	
80  }
