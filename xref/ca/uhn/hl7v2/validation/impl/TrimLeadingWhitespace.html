26  package ca.uhn.hl7v2.validation.impl;
27  
28  import java.util.regex.Pattern;
29  
30  import ca.uhn.hl7v2.validation.ValidationException;
31  import ca.uhn.hl7v2.validation.builder.PrimitiveRuleBuilder;
32  
33  /**
34   * Performs no validation but removes leading whitespace in the correct() method.
35   * 
36   * @author Bryan Tripp
37   * @author Christian Ohr
38   * 
39   * @deprecated use {@link PrimitiveRuleBuilder#leftTrim()} instead
40   */
41  @SuppressWarnings("serial")
42  public class TrimLeadingWhitespace extends AbstractPrimitiveTypeRule {
43  
44  	private static final Pattern LEADING_WHITESPACE = Pattern.compile("^\\s+");
45  
46  	/**
47  	 * Removes leading whitespace.
48  	 * 
49  	 * @see ca.uhn.hl7v2.validation.PrimitiveTypeRule#correct(java.lang.String)
50  	 */
51  	public String correct(String value) {
52  		return value != null ? LEADING_WHITESPACE.matcher(value).replaceAll("") : null;
53  	}
54  
55  	public ValidationException[] apply(String value) {
56  		return passed();
57  	}
58  
59  	/**
60  	 * @see ca.uhn.hl7v2.validation.Rule#getDescription()
61  	 */
62  	public String getDescription() {
63  		return "Leading whitespace removed";
64  	}
65  
66  }
