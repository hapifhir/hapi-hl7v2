26  package ca.uhn.hl7v2.validation.builder.support;
27  
28  import java.util.ArrayList;
29  import java.util.List;
30  
31  import ca.uhn.hl7v2.ErrorCode;
32  import ca.uhn.hl7v2.Location;
33  import ca.uhn.hl7v2.model.Message;
34  import ca.uhn.hl7v2.validation.MessageRule;
35  import ca.uhn.hl7v2.validation.ValidationException;
36  import ca.uhn.hl7v2.validation.impl.AbstractMessageRule;
37  
38  /**
39   * Validation rule for blaming a wring HL7 version
40   * 
41   * @author Christian Ohr
42   */
43  @SuppressWarnings("serial")
44  public class WrongVersionRule extends AbstractMessageRule {
45  
46  	public static final MessageRule WRONG_VERSION = new WrongVersionRule();
47  	
48  	public ValidationException[] apply(Message msg) {
49  		List<ValidationException> exceptions = new ArrayList<ValidationException>();
50  
51  		ValidationException ve = new ValidationException("Invalid version: " + msg.getVersion());
52  		Location location = new Location()
53  		    .withSegmentName("MSH")
54  		    .withField(12);
55  		ve.setLocation(location);
56  		ve.setError(ErrorCode.UNSUPPORTED_VERSION_ID);
57  		exceptions.add(ve);
58  
59  		return exceptions.toArray(new ValidationException[exceptions.size()]);
60  	}
61  
62  }
