3   import static ca.uhn.hl7v2.hoh.util.StringUtils.*;
4   
5   import java.io.UnsupportedEncodingException;
6   import java.nio.charset.StandardCharsets;
7   
8   import ca.uhn.hl7v2.hoh.util.VersionLogger;
9   import ca.uhn.hl7v2.hoh.util.repackage.Base64;
10  
11  public class Hl7OverHttpRequestEncoder extends AbstractHl7OverHttpEncoder {
12  
13  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(Hl7OverHttpRequestEncoder.class);
14  	private String myHost;
15  	private int myPort;
16  
17  	public void setHost(String theHost) {
18  		myHost = theHost;
19  	}
20  
21  	public void setPort(int thePort) {
22  		myPort = thePort;
23  	}
24  	private final boolean myAcceptGzip = false;
25  
26  	@Override
27  	protected void addSpecificHeaders() {
28  		if (isNotBlank(getUsername()) && isNotBlank(getPassword())) {
29  			String authorizationUnescaped = defaultString(getUsername()) + ":" + defaultString(getPassword());
30  			String encoded;
31              encoded = Base64.encodeBase64String(authorizationUnescaped.getBytes(StandardCharsets.ISO_8859_1));
32              getHeaders().put("Authorization", "Basic " + encoded);
33  		}
34  
35  		if (myAcceptGzip) {
36  			getHeaders().put("Accept-Encoding", "gzip");
37  		}
38  		getHeaders().put("User-Agent", "HAPI (HL7 over HTTP) Client " + VersionLogger.getVersion());
39  
40  		StringBuilder hostBuilder = new StringBuilder();
41  		if (isNotBlank(myHost)) {
42  			hostBuilder.append(myHost);
43  			if (myPort > 0) {
44  				hostBuilder.append(":");
45  				hostBuilder.append(myPort);
46  			} else {
47  				ourLog.warn("Host has been set, but port has not");
48  			}
49  		} else {
50  			ourLog.warn("Host has not been set");
51  		}
52  		getHeaders().put("Host", hostBuilder.toString());
53  	}
54  
55  	@Override
56  	protected void setActionLineAppropriately() {
57  		setActionLine("POST " + getPath() + " HTTP/1.1");
58  	}
59  
60  }
