026package ca.uhn.hl7v2.validation.builder.support;
027
028import java.util.ArrayList;
029import java.util.List;
030import java.util.Set;
031
032import ca.uhn.hl7v2.HL7Exception;
033import ca.uhn.hl7v2.model.Message;
034import ca.uhn.hl7v2.model.Structure;
035import ca.uhn.hl7v2.model.SuperStructure;
036import ca.uhn.hl7v2.validation.MessageRule;
037import ca.uhn.hl7v2.validation.ValidationException;
038import ca.uhn.hl7v2.validation.impl.AbstractMessageRule;
039
040/**
041 * Validation rule for detecting populated elements in a {@link SuperStructure}
042 * which do not belong to the given message type
043 * 
044 * @author James Agnew
045 */
046@SuppressWarnings("serial")
047public class OnlyAllowableSegmentsInSuperstructureRule extends AbstractMessageRule {
048
049        public static final MessageRule ONLY_ALLOWABLE_SEGMENTS = new OnlyAllowableSegmentsInSuperstructureRule();
050        
051        public ValidationException[] apply(Message msg) {
052                List<ValidationException> exceptions = new ArrayList<ValidationException>();
053                
054                if (msg instanceof SuperStructure) {
055                        checkStructure((SuperStructure)msg, exceptions);
056                }
057                
058                return exceptions.toArray(new ValidationException[exceptions.size()]);
059        }
060
061        private void checkStructure(SuperStructure theMsg, List<ValidationException> theExceptions) {
062                String messageStructure = theMsg.getMessage().getName();
063                Set<String> nonStandardNames = theMsg.getNonStandardNames();
064
065                FORNAME:
066                for (String name : theMsg.getNames()) {
067                        if (nonStandardNames != null && nonStandardNames.contains(name)) {
068                                continue;
069                        }
070                        
071                        try {
072                                for (Structure rep : theMsg.getAll(name)) {
073                                        
074                                        if (!rep.isEmpty()) {
075                                                Set<String> structuresWhichChildAppliesTo = theMsg.getStructuresWhichChildAppliesTo(name);
076                                                if (!structuresWhichChildAppliesTo.contains(messageStructure)) {
077                                                        String msgSimpleName = theMsg.getMessage().getClass().getSimpleName();
078                                                        theExceptions.add(new ValidationException("Message (superstructure " + msgSimpleName + ") of type " + messageStructure + " must not have content in " + name));
079                                                }
080                                                continue FORNAME;
081                                        }
082                                        
083                                }
084                        } catch (HL7Exception e) {
085                                // should not happen
086                                throw new Error("Can't get rep of structure " + name + ". This is probably a HAPI bug");
087                        }
088                        
089                }
090        }
091
092}
