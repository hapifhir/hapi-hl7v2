027package ca.uhn.hl7v2.conf.store;
028
029import ca.uhn.hl7v2.conf.ProfileException;
030
031/**
032 * Interface used retrieving and validating codes from user defined and HL7 specific tables that
033 * correspond to a conformance profile.
034 * 
035 * @author Neal Acharya
036 */
037public interface CodeStore {
038
039    /**
040     * Retrieves all codes for a given conformance profile and codeSystem.
041     * 
042     * @param codeSystem a table of codes (for example, HL70001 for administrative sex) valid tables
043     *            are defined in the HL7 table 0396
044     * @return a list of valid codes
045     * @throws ProfileException
046     */
047    public String[] getValidCodes(String codeSystem) throws ProfileException;
048
049    /**
050     * Validates the codeSystem against the input conformance profile. If valid then output is
051     * 'true' else 'false'.
052     * 
053     * @param codeSystem
054     * @return <code>true</code> if CodeStore knows the codeSystem
055     */
056    public boolean knowsCodes(String codeSystem);
057
058    /**
059     * Validates the input code value against the input conformance profile and corresponding input
060     * codeSystem. Returns true if the code is valid and false if it isn't.
061     * 
062     * @param codeSystem
063     * @param code
064     * @return <code>true</code> if code is valid for the codeSystem
065     */
066    public boolean isValidCode(String codeSystem, String code);
067
068}
