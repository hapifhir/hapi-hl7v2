028package ca.uhn.hl7v2.model;
029
030import java.util.*;
031
032/**
033 * This class is used to provide utility functions for other datatype classes and methods.
034 */
035
036public class DataTypeUtil {
037
038    public DataTypeUtil() {
039    }
040
041
042    /**
043     * Preappend the zeros to the beginning of num such that the total length
044     * equals totalDigitLength and return the string representation of the new number
045     *
046     * @param num              number
047     * @param totalDigitLength number of digits
048     * @return string representation of the new number
049     */
050    public static String preAppendZeroes(int num, int totalDigitLength) {
051        return (totalDigitLength == 0) ?
052                Integer.toString(num) :
053                String.format(String.format("%%0%dd", totalDigitLength), num);
054    }
055
056
057    /**
058     * Return a signed four digit integer indicating the local
059     * GMT offset. This is the HL7 Offset format in integer representation.
060     *
061     * @return a signed four digit integer indicating the local GMT offset
062     */
063    public static int getLocalGMTOffset() {
064        GregorianCalendar currentTime = new GregorianCalendar();
065        int gmtOffSet = currentTime.get(GregorianCalendar.ZONE_OFFSET);
066        int offSetSignInt = gmtOffSet < 0 ? -1 : 1;
067        //get the absolute value of the gmtOffSet
068        int absGmtOffSet = Math.abs(gmtOffSet);
069        int gmtOffSetHours = absGmtOffSet / (3600 * 1000);
070        int gmtOffSetMin = (absGmtOffSet / 60000) % (60);
071        // return the offset value HL7 format
072        return ((gmtOffSetHours * 100) + gmtOffSetMin) * offSetSignInt;
073    }
074
075
076}
