028package ca.uhn.hl7v2.concurrent;
029
030import java.util.concurrent.CompletionService;
031import java.util.concurrent.ExecutorCompletionService;
032import java.util.concurrent.ExecutorService;
033import java.util.concurrent.Executors;
034import java.util.concurrent.ThreadFactory;
035import java.util.concurrent.atomic.AtomicInteger;
036
037/**
038 * Provides a global cached thread pool if Connections and Servers are not
039 * initialized with their specific {@link ExecutorService} instances.
040 */
041public class DefaultExecutorService {
042
043        static ExecutorService defaultExecutorService;
044
045        public static <V> CompletionService<V> completionService(ExecutorService executor) {
046                return new ExecutorCompletionService<V>(executor);
047        }
048
049        public static synchronized ExecutorService getDefaultService() {
050                if (defaultExecutorService == null || defaultExecutorService.isShutdown()) {
051                        defaultExecutorService = Executors.newCachedThreadPool(new MyThreadFactory());
052                }
053                return defaultExecutorService;
054        }
055
056        /**
057         * @return true if the service is not null and was created by this class
058         */
059        public static boolean isDefaultService(ExecutorService service) {
060                return service != null && service == defaultExecutorService;
061        }
062
063        /**
064         * @see {@link ExecutorService#shutdown()}
065         */
066        public static void shutdown() {
067                defaultExecutorService.shutdown();
068        }
069
070        /**
071         * @see {@link ExecutorService#shutdownNow()}
072         */
073        public static void shutdownNow() {
074                defaultExecutorService.shutdownNow();
075        }
076
077        private static class MyThreadFactory implements ThreadFactory {
078
079                private ThreadGroup group;
080                private AtomicInteger threadNum;
081
082                public MyThreadFactory() {
083                        group = Thread.currentThread().getThreadGroup();
084                        threadNum = new AtomicInteger(1);
085                }
086
087                public Thread newThread(Runnable theR) {
088                        String name = "hapi-worker-" + threadNum.getAndIncrement();
089                        Thread t = new Thread(group, theR, name, 0);
090                        if (t.isDaemon()) {
091                                t.setDaemon(false);
092                        }
093                        if (t.getPriority() != Thread.NORM_PRIORITY) {
094                                t.setPriority(Thread.NORM_PRIORITY);
095                        }
096                        return t;
097                }
098
099        }
100
101}
